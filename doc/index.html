Image uploader

Example

            var uploader = new ImageUploader('[data-js="uploader"]', {
                maxItems: 2
            });
            uploader.on('adding', function(file, done, progress) {
                progress(10);
                var croppedFile = crop(file, 200, 200);
                progress(50);
                var picture = {
                    url: croppedFile
                };
                progress(100);
                done(picture);
            });
            uploader.add({
                url: 'http://example.com/picture.jpg'
            });

Options

maxItems = int
int Max number of images
ajax.add = function(file, callback, progress)
override ajax query to add picture
ajax.update = function(picture, file, callback, progress)
override ajax query to update picture
ajax.delete = function(photo, callback)
override ajax query to delete picture
